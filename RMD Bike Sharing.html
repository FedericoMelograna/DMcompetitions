# get the required R packages

library(caret)
library(rpart)
library(rattle)
library(rpart.plot)
library(RColorBrewer)
library(randomForest)
library(fasttime)
library(lubridate)
# import data
train <- read.csv("http://bee-fore.s3-eu-west-1.amazonaws.com/datasets/99.csv")
test <- read.csv("http://bee-fore.s3-eu-west-1.amazonaws.com/datasets/100.csv")

test$registered=NA;test$casual=NA;test$count=NA
all=rbind(train,test)

all$holiday=as.factor(all$holiday)
all$workingday=as.factor(all$workingday)
all$hour=hour(fastPOSIXct(all$datetime,"GMT"))
all$hour=as.factor(all$hour)
date=substr(all$datetime,1,10)
days<-weekdays(as.Date(date))
all$day=days
all$year=year(fastPOSIXct(all$datetime,"GMT"))
all$year=as.factor(all$year)
all$hour=as.integer(all$hour)

all$month=month(fastPOSIXct(all$datetime,"GMT"))
all$month=as.integer(all$month)
train=all[1:10886,]
test=all[10887:17379,]

houra_reg=rpart(registered~hour,data=train)
fancyRpartPlot(houra_reg)

houra_cas=rpart(casual~hour,data=train)
fancyRpartPlot(houra_cas)

tempareg=rpart(registered~temp,data=train)
fancyRpartPlot(tempareg)

tempacas=rpart(casual~temp,data=train)
fancyRpartPlot(tempacas)

all=rbind(train,test)
#bin ore nei due casi 
all$reg_hour_bin=0
all$reg_hour_bin[all$hour<8]=1
all$reg_hour_bin[all$hour>=22]=2
all$reg_hour_bin[all$hour>9 & all$hour<18]=3
all$reg_hour_bin[all$hour==8]=4
all$reg_hour_bin[all$hour==9]=5
all$reg_hour_bin[all$hour==20 | all$hour==21]=6
all$reg_hour_bin[all$hour==19 | all$hour==18]=7
all$cas_hour_bin=0
all$cas_hour_bin[all$hour<=8]=1
all$cas_hour_bin[all$hour==9 |all$hour==10]=2
all$cas_hour_bin[all$hour>=11 & all$hour<=21]=3
all$cas_hour_bin[all$hour>21]=4

#bin tempo
all$temp_reg=1
all$temp_reg[all$temp>=13 & all$temp<23]=2
all$temp_reg[all$temp>=23 & all$temp<30]=3
all$temp_reg[all$temp>=30]=4
all$temp_cas=1
all$temp_cas[all$temp>=15 & all$temp<23]=2
all$temp_cas[all$temp>=23 & all$temp<30]=3
all$temp_cas[all$temp>=30]=4
#binning anno per serie storica

all$year_part=ifelse(all$year=='2011' & all$month<=6,1,
                     ifelse(all$year=='2011' & all$month>6,2,
                            ifelse(all$year=='2012' & all$month<=6,3,4)))



all$type_day<- ifelse(all$holiday==0 & all$workingday==0, "weekend",
                  ifelse(all$holiday==1,"holiday",
                         ifelse(all$holiday==0 & all$workingday==1,"working day",0)))

all$weekend=ifelse(all$day=="sabato" | all$day=="domenica",1,0)


#FACTOR

all$season=as.factor(all$season)
all$holiday=as.factor(all$holiday)
all$workingday=as.factor(all$workingday)
all$weather=as.factor(all$weather)
all$hour=as.factor(all$hour)
all$month=as.factor(all$month)
all$type_day=as.factor(all$type_day)
all$day=as.factor(all$day)

train=all[1:10886,]
test=all[10887:17379,]
train$regr_log1=(log(train$registered+1)) 
train$cas_log1=log1p(train$casual)
#Modello
set.seed(123)
fit1 <- randomForest(regr_log1 ~ hour +workingday+day+holiday+ type_day +temp_reg+humidity+atemp+windspeed+season+weather+reg_hour_bin+weekend+year+year_part, data=train,importance=TRUE, ntree=300)

pred1=predict(fit1,test)

set.seed(123)
fit2 <- randomForest(cas_log1 ~hour + type_day+day+humidity+atemp+temp_cas+windspeed+season+weather+holiday+workingday+cas_hour_bin+weekend+year+year_part, data=train,importance=TRUE, ntree=300)

pred2=predict(fit2,test)

#file finale
finale=expm1(pred1)+expm1(pred2)
